if (GENERATE_TYPES)
  generate_f90_types(src/ExtLoadsDX_Registry.txt ${CMAKE_CURRENT_LIST_DIR}/src/ExtLoadsDX_Types.f90 -ccode)
  generate_f90_types(src/ExtLoads_Registry.txt ${CMAKE_CURRENT_LIST_DIR}/src/ExtLoads_Types.f90)
endif()

# copy the header files to their build location
configure_file(src/ExtLoadsDX_Types.h ${CMAKE_CURRENT_BINARY_DIR} COPYONLY)

set(EXTLOADS_LIBS_SOURCES
  src/ExtLoads.f90
  # Autogenerated files
  src/ExtLoads_Types.f90
  src/ExtLoadsDX_Types.f90
  )

add_library(extloadslib ${EXTLOADS_LIBS_SOURCES})
target_link_libraries(extloadslib nwtclibs beamdynlib)

install(TARGETS extloadslib
  EXPORT "${CMAKE_PROJECT_NAME}Libraries"
  RUNTIME DESTINATION bin
  LIBRARY DESTINATION lib
  ARCHIVE DESTINATION lib)

install(FILES
  ${CMAKE_CURRENT_BINARY_DIR}/ExtLoadsDX_Types.h
  DESTINATION include)
